Circuit 1

Design a circuit with two LEDs that turn on and off alternately every 3 seconds.

int led1 = 9;
    int led2 = 10;
    void setup() {
        pinMode(led1, OUTPUT);
        pinMode(led2, OUTPUT);
    }
    void loop() {
        digitalWrite(led1, HIGH);
        digitalWrite(led2, LOW);
        delay(3000);
        digitalWrite(led1, LOW);
        digitalWrite(led2, HIGH);
        delay(3000);
    }

Circuit 2

Design a circuit with a push button and one LED that turns on when the button is pressed.

int button = 2;
    int led = 9;
    void setup() {
        pinMode(button, INPUT);
        pinMode(led, OUTPUT);
    }
    void loop() {
        if (digitalRead(button) == HIGH) {
            digitalWrite(led, HIGH);
        } else {
            digitalWrite(led, LOW);
        }
    }

Circuit 3

Design a circuit with an ultrasonic sensor and an LED that turns on when an object is detected within 20cm.

#include <NewPing.h>
    #define TRIG 7
    #define ECHO 6
    #define LED 9
    NewPing sonar(TRIG, ECHO, 200);
    void setup() {
        pinMode(LED, OUTPUT);
        Serial.begin(9600);
    }
    void loop() {
        int distance = sonar.ping_cm();
        if (distance > 0 && distance <= 20) {
            digitalWrite(LED, HIGH);
        } else {
            digitalWrite(LED, LOW);
        }
        delay(500);
    }

Circuit 4

Design a circuit with an ultrasonic sensor that prints "Object Detected" on the serial monitor when an object is detected within 20cm.

#include <NewPing.h>
    #define TRIG 7
    #define ECHO 6
    NewPing sonar(TRIG, ECHO, 200);
    void setup() {
        Serial.begin(9600);
    }
    void loop() {
        int distance = sonar.ping_cm();
        if (distance > 0 && distance <= 20) {
            Serial.println("Object Detected");
        }
        delay(500);
    }
